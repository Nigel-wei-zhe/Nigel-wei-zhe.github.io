Sass - @for, @each & map, HSL Functions

jot down notes

@for

@for $i from 1 to 3 {   // to: 1~3 不包含3, through: 1~3 包含3
    .box-#{ $i } {      // classname使用變數要加上 '#{}'
        width: $i;
    }
}

OUTPUT:

.box-1 {
    width: 1;
}
.box-2 {
    width: 2;
}

@each & map

$themes: {
    'primary': blue,
    'danger': red,
    'secondary': orange
}

@each $key, $value in $themes {
  .box-#{ $key } {
    background-color: $value;
  }
}

OUTPUT:

.box-primary {
  background-color: blue;
}
.box-danger {
  background-color: red;
}
.box-secondary {
  background-color: orange;
}
